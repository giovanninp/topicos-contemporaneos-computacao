CREATE TABLE competitor(
	id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	name VARCHAR(50) NOT NULL,
	registration BIGINT NOT NULL,
	performance FLOAT NOT NULL,
	standard FLOAT NOT NULL
);

CREATE TABLE bet(
	id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	negotiator_register VARCHAR(120) NOT NULL,
	date DATETIME NOT NULL,
	number BIGINT NOT NULL UNIQUE
);

CREATE TABLE ticket (
	id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	base_value FLOAT NOT NULL,
	comission FLOAT NOT NULL,
	paid_value FLOAT NOT NULL,
	bet_id INT(6) UNSIGNED NOT NULL,
	competitor_id INT(6) UNSIGNED NOT NULL,
	FOREIGN KEY(bet_id) REFERENCES bet(id),	
	FOREIGN KEY(competitor_id) REFERENCES competitor(id)
);
